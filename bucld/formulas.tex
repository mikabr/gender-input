$$
B(k; n, p)
$$

$$
p_1 = \frac{k_1}{n_1}
\quad
p_2 = \frac{k_2}{n_2}
\quad
p = \frac{k_1 + k_2}{n_1 + n_2}
$$

$$
\Lambda = \frac{B(k_1; n_1, p) B(k_2; n_2, p)}{B(k_1; n_1, p_1) B(k_2; n_2, p_2)}
$$


$$
-2 \log{\Lambda} \sim \chi^{2}
$$
